class Solution:
    def findRelativeRanks(self, score: List[int]) -> List[str]:
        rank=sorted(score , reverse=True)
        f={}
        for i in range(len(score)):
            if i ==0:
                f[rank[i]]="Gold Medal"
            if i ==1:
                f[rank[i]]="Silver Medal"
            if i ==2:
                f[rank[i]]="Bronze Medal"
            elif i>2:
                f[rank[i]]=str(i+1)
        
        ans=[]
        for i in score:
            ans.append(f[i])
        return ans
