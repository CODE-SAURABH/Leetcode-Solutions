# BruteForce Method

```python
class Solution:
    def findTheDistanceValue(self, arr1: List[int], arr2: List[int], d: int) -> int:
        distance_value = 0
        
        for i in arr1:
            is_valid = True
            for j in arr2:
                if abs(i - j) <= d:
                    is_valid = False
                    break
            if is_valid:
                distance_value += 1
        
        return distance_value

# Example usage
solution = Solution()
arr1 = [4, 5, 8]
arr2 = [10, 9, 1, 8]
d = 2
print(solution.findTheDistanceValue(arr1, arr2, d))  # Expected output: 2

```

## Optimize Method

```python
class Solution:
    def findTheDistanceValue(self, arr1, arr2, d):
        arr2.sort()

        count = 0
        for i in range(len(arr1)):
            s = arr1[i] - d
            e = arr1[i] + d
            left, right = 0, len(arr2)-1
            while left <= right:
                mid = (left + right) // 2
                if arr2[mid] >= s and arr2[mid] <= e:
                    break
                elif arr2[mid] < s:
                    left = mid +1
                else:
                    right = mid - 1

            if left > right:
                count +=1
        return count
```
