# BruteForce Method

```python

class Solution:
    def isPrefixOfWord(self, sentence: str, searchWord: str) -> int:
        words = sentence.split(" ")  
        for i in range(len(words)):
            
            if words[i][:len(searchWord)] == searchWord:
                return i + 1  
        return -1  


# Optimize Method

```python
class Solution:
    def isPrefixOfWord(self, sentence: str, searchWord: str) -> int:
        i = 0  
        j = 0  
        word_index = 1 

        while j < len(sentence):
            if sentence[j] == ' ':
                i = j + 1
                word_index += 1
            elif sentence[i:i + len(searchWord)] == searchWord: 
                return word_index
            j += 1

        return -1
