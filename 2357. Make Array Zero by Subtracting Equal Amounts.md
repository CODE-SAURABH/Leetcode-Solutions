# BruteForce Method

```python

class Solution:
    def minimumOperations(self, nums: List[int]) -> int:
        if sum(nums) == 0:
            return 0
        nums.sort()
        count = 0
        while sum(nums) != 0:
            mini = float('inf')
            for i in nums:
                if i != 0:
                    mini = i
                    break
            nums = [i - mini if i > 0 else i for i in nums]
            count += 1
        return count

# Optimize Method

class Solution:
    def minimumOperations(self, nums: List[int]) -> int:
        nums = list(set(nums)) 
        nums.sort()  
        count = 0
        for num in nums:
            if num > 0:  
                count += 1
        
        return count
