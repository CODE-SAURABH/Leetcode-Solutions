# BruteForce Method

```python
class Solution:
    def intersection(self, nums: List[List[int]]) -> List[int]:
        l=[]
        for i in nums:
            l.extend(i)
        f={}
        for i in l:
            if i not in f:
                f[i]=1
            else:
                f[i]+=1
        n=len(nums)
        ans=[]
        for k,v in f.items():
            if v==n:
                ans.append(k)
        
        ans.sort()
        return ans


# Optimize Method

```python

class Solution:
    def intersection(self, nums: List[List[int]]) -> List[int]:
        result = set(nums[0])
        for arr in nums[1:]:
            result &= set(arr)        
        return sorted(result)
