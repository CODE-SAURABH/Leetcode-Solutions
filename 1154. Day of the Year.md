class Solution:
    def dayOfYear(self, date: str) -> int:
        days_per_month = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]
        year, month, day = map(int, date.split("-"))

        if (year % 4 == 0 and year % 100 != 0) or year % 400 == 0:
            days_per_month[1] = 29

        days_number = day + sum(days_per_month[:month - 1])
        return days_number
