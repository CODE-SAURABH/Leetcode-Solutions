# BruteForce Method

```python
class Solution:
    def countDigitOne(self, n: int) -> int:
        c=0
        for i in range(n+1):
            c=c+int(str(i).count('1'))
        return c

        

# Optimize Method

```python

class Solution:
    def countDigitOne(self,n: int) -> int:
        countr = 0
        i = 1
        while i <= n:
            divider = i * 10
            countr += (n // divider) * i + min(max(n % divider - i + 1, 0), i)
            i *= 10
    
        return countr
