# BruteForce Method

```python
class Solution:
    def smallestNumber(self, n: int, t: int) -> int:
        def product_of_digits(num):
            product = 1
            for digit in str(num):
                product *= int(digit)
            return product
        if t==1:
            return n
        for i in range(n, 101):
            if product_of_digits(i) % t == 0:
                return i
```

## Optimize Method

```python
class Solution:
    def smallestNumber(self, n: int, t: int) -> int:
        if n == 100:
            return n

        while n <= 100:
            if n < 10:
                if n % t == 0:
                    return n
            else:
                d = str(n)
                l = int(d[0])
                r = int(d[1])
                p = l * r
                if p % t == 0:
                    return n
            n += 1
        
```
