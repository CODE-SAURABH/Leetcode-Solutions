class Solution:
    def findThePrefixCommonArray(self, A: List[int], B: List[int]) -> List[int]:
        s = 0
        l = []
        f = {}

        for i in range(len(A)):
        
            if A[i] not in f:
                f[A[i]] = 0
            if B[i] not in f:
                f[B[i]] = 0

            f[A[i]] += 1
            f[B[i]] += 1

            if f[A[i]] == 2:
                s += 1
            if f[B[i]] == 2 and A[i] != B[i]:
                s += 1

            l.append(s)

        return l
