# BruteForce Method

```python
class Solution:
    def kthSmallestPrimeFraction(self, arr: List[int], k: int) -> List[int]:
        f={}
        i=0
        l=[]
        while i<len(arr)-1:
            j=i+1
            while j<len(arr):
                l.append(arr[i]/arr[j])
                if (arr[i]/arr[j]) not in f:
                    f[arr[i]/arr[j]]=[arr[i],arr[j]]
                else:
                    f[arr[i]/arr[j]]=[f[arr[i]/arr[j]],[arr[i],arr[j]]]
                j+=1
            i+=1
        l.sort()
        t=l[k-1]
        return f[t]

```
