class Solution:
    def minElement(self, nums: List[int]) -> int:
        def digit_sum(num):
            total = 0
            while num > 0:
                total += num % 10
                num //= 10
            return total
        
        min_val = float('inf')
        for num in nums:
            min_val = min(min_val, digit_sum(num))
        
        return min_val

class Solution:
    def minElement(self, nums: List[int]) -> int:
        for i in range(len(nums)):
            nums[i]=sum([int(d) for d in str(nums[i])])
        return min(nums)
